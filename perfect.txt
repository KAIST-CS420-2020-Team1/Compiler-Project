/*
A perfect number is an integer which is equal to the sum of all its divisors including 1 but excluding the number itself.
*/

//const int max = 500;

void main()
{
	int max;
	max = 100;
	int i, j, k;
	int rem, sum; //rem : remainder
 	for (i = 2; i < max; i++) {
 		sum = 0;
 		k = i / 2;
 		for (j = 1; j < k + 1; j++) {
 			rem = i - (i / j) * j;
 			if (rem == 0) sum += j;
 		}
 		if (i == sum) printf("%d\n" , i);
 	}
	return i;
}
